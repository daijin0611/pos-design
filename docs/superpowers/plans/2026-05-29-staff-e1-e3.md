# 员工端 E1-E3 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在现有 pos-h5 Vue3 + Vant 项目中，替换旧的员工端页面，实现信息流式首页 + E1 工资查询 + E2 回访提醒 + E3 生日提醒，共 7 个页面。

**Architecture:** 卡片式信息流首页（无底部 Tab），功能页面通过 NavBar 返回导航。Mock 数据存放在 `src/mock/staff.js`，通过 Pinia store 消费。组件按职责拆分为独立文件，Vant 组件通过 auto-import 自动注册。

**Tech Stack:** Vue 3 Composition API + Vant 4 + Vue Router 4 (hash mode) + Pinia

// src/utils/format.js

/**
 * 格式化金额
 * @param {number} amount
 * @param {boolean} withSymbol - 是否带 ¥ 符号
 * @returns {string}
 */
export function formatMoney(amount, withSymbol = true) {
  const formatted = amount.toLocaleString('zh-CN', {
    minimumFractionDigits: 2,
    maximumFractionDigits: 2,
  })
  return withSymbol ? `¥ ${formatted}` : formatted
}

/**
 * 格式化月份显示
 * @param {string} monthStr - 'YYYY-MM'
 * @returns {string} '2026年5月'
 */
export function formatMonth(monthStr) {
  const [year, month] = monthStr.split('-')
  return `${year}年${parseInt(month)}月`
}

/**
 * 格式化日期为友好显示
 * @param {string} dateStr - 'YYYY-MM-DD'
 * @returns {string} '5月18日'
 */
export function formatDate(dateStr) {
  const date = new Date(dateStr)
  return `${date.getMonth() + 1}月${date.getDate()}日`
}

/**
 * 获取问候语
 * @returns {{ text: string, emoji: string }}
 */
export function getGreeting() {
  const hour = new Date().getHours()
  if (hour < 6) return { text: '凌晨好', emoji: '🌙' }
  if (hour < 12) return { text: '早上好', emoji: '☀️' }
  if (hour < 14) return { text: '中午好', emoji: '🌤' }
  if (hour < 18) return { text: '下午好', emoji: '⛅' }
  return { text: '晚上好', emoji: '🌙' }
}

/**
 * 格式化生日倒计时
 * @param {number} days
 * @returns {string}
 */
export function formatDaysUntil(days) {
  if (days === 0) return '今天'
  if (days === 1) return '明天'
  if (days === 2) return '后天'
  return `${days}天后`
}
```

// src/stores/staff.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import {
  staffInfo,
  salaryData,
  remindersData,
  birthdaysData,
  memberProfiles,
} from '../mock/staff'

export const useStaffStore = defineStore('staff', () => {
  // --- 员工信息 ---
  const info = ref(staffInfo)

  // --- 工资 ---
  const currentMonth = ref('2026-05')
  const salaryMonths = computed(() =>
    Object.keys(salaryData).sort().reverse()
  )
  const currentSalary = computed(
    () => salaryData[currentMonth.value] || null
  )

  function setMonth(month) {
    currentMonth.value = month
  }

  function getSalaryDetail(category) {
    const salary = salaryData[currentMonth.value]
    if (!salary) return []
    return salary.details[category] || []
  }

  // --- 回访提醒 ---
  const reminders = ref([...remindersData])

  const pendingReminders = computed(() =>
    reminders.value.filter((r) => r.status === 'pending')
  )
  const overdueReminders = computed(() =>
    reminders.value.filter((r) => r.status === 'overdue')
  )
  const completedReminders = computed(() =>
    reminders.value.filter((r) => r.status === 'completed')
  )

  /** 首页展示：超期 + 待跟进，最多3条 */
  const topReminders = computed(() => {
    const overdue = overdueReminders.value
    const pending = pendingReminders.value
    return [...overdue, ...pending].slice(0, 3)
  })

  function completeReminder(id, note = '') {
    const item = reminders.value.find((r) => r.id === id)
    if (item) {
      item.status = 'completed'
      item.overdueDays = 0
      item.completedAt = new Date().toLocaleString('zh-CN')
      item.note = note
    }
  }

  // --- 生日提醒 ---
  const birthdays = ref(birthdaysData)

  const thisWeekBirthdays = computed(() =>
    birthdays.value.filter((b) => b.section === 'thisWeek')
  )
  const thisMonthBirthdays = computed(() =>
    birthdays.value.filter((b) => b.section === 'thisMonth')
  )

  // --- 会员档案 ---
  function getMemberProfile(memberId) {
    return memberProfiles[memberId] || null
  }

  // --- 统计 ---
  const todayTodoCount = computed(() => {
    const overdue = overdueReminders.value.length
    const pending = pendingReminders.value.length
    const birthday = thisWeekBirthdays.value.length
    return overdue + pending + birthday
  })

  const todayDoneCount = computed(() => completedReminders.value.length)

  return {
    info,
    currentMonth,
    salaryMonths,
    currentSalary,
    setMonth,
    getSalaryDetail,
    reminders,
    pendingReminders,
    overdueReminders,
    completedReminders,
    topReminders,
    completeReminder,
    birthdays,
    thisWeekBirthdays,
    thisMonthBirthdays,
    getMemberProfile,
    todayTodoCount,
    todayDoneCount,
  }
})
```

<script setup>
import { ref } from 'vue'

const props = defineProps({
  show: { type: Boolean, default: false },
  memberName: { type: String, default: '' },
  reminderId: { type: Number, default: 0 },
})

const emit = defineEmits(['update:show', 'confirm'])

const note = ref('')
const selectedTags = ref([])

const quickTags = ['效果好', '需继续', '已预约下次', '不满意']

function toggleTag(tag) {
  const idx = selectedTags.value.indexOf(tag)
  if (idx === -1) {
    selectedTags.value.push(tag)
    note.value = note.value
      ? `${note.value}，${tag}`
      : tag
  } else {
    selectedTags.value.splice(idx, 1)
  }
}

function isTagSelected(tag) {
  return selectedTags.value.includes(tag)
}

function confirm() {
  emit('confirm', {
    id: props.reminderId,
    note: note.value,
  })
  note.value = ''
  selectedTags.value = []
}

function onClose() {
  emit('update:show', false)
}
</script>

<template>
  <van-action-sheet
    :model-value="show"
    @update:model-value="$emit('update:show', $event)"
